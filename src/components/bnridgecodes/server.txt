const express = require("express");
const PlataformaFactory = require("./factories/PlataformaFactory");
const NotificacionFactory = require("./factories/NotificacionFactory");

const app = express();
app.use(express.json());

app.post("/api/notificaciones", (req, res) => {
  try {
    const { tipo, plataforma, titulo, mensaje } = req.body;

    const plataformaObj = PlataformaFactory.crear(plataforma);
    const notificacion = NotificacionFactory.crear(tipo, plataformaObj);

    const resultado = notificacion.enviar(titulo, mensaje);

    res.status(200).json({
      success: true,
      data: resultado
    });

  } catch (error) {
    res.status(400).json({
      success: false,
      error: error.message
    });
  }
});

const PORT = 3000;
app.listen(PORT, () => {
  console.log(`Servidor corriendo en puerto ${PORT}`);
});
